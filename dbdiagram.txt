
Table user as U {
  id int [pk, increment] // auto-increment
  param_id int [ref: > param.id]
  name varchar
}

Table cannon as Can {
   id int [pk, increment] // auto-increment
   desc varchar 
   weight decimal
   price decimal
   length decimal
}

Table munition as Mun {
   id int [pk, increment] // auto-increment
   desc varchar 
   weight decimal [note: 'poids du chargeur plein']
   price decimal
   bullet_weight decimal [note: 'poids d\'une balle']
   cx decimal [note: 'coefficient de trainée']
}

Table plateform as Pla {
   id int [pk, increment] // auto-increment
   desc varchar 
   weight decimal
   price decimal
   length decimal
   standard_cannon_length decimal
}

Table config as C {
   id int [pk, increment] // auto-increment
   name varchar 
   cannon_id int [ref: > Can.id]
   munition_id int [ref: > Mun.id]
   plateform_id int [ref: > Pla.id]
}

Table param as P {
   id int [pk, increment] // auto-increment
   //user_id int [ref: - U.id]
   name varchar 
   height decimal
   angle decimal [note: 'radian'] 
}

Table config_user as CU {
   config_id int [ref: > C.id]
   user_id int [ref: > U.id]
    Indexes {
    (config_id, user_id) [pk]
  }
}

Table init_speed {
   cannon_id int [ref: > Can.id]
   munition_id int [ref: > Mun.id]
   init_speed decimal
    Indexes {
    (cannon_id, munition_id) [pk]
  }
}
